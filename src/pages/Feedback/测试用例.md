## 自动化测试代码

```typescript
// 可以在 Playwright 或 Cypress 中运行

import { test, expect } from '@playwright/test'

test.describe('反馈页面图片上传', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/feedback')
  })

  test('单张图片上传', async ({ page }) => {
    // 填写表单
    await page.selectOption('select', 'bug')
    await page.fill('textarea', '这是一个测试反馈')
    
    // 上传图片
    const fileInput = page.locator('input[type="file"]')
    await fileInput.setInputFiles('test-image.jpg')
    
    // 验证缩略图显示
    const thumbnail = page.locator('.ant-upload-list-picture-card .ant-upload-list-item')
    await expect(thumbnail).toBeVisible()
  })

  test('最多上传 5 张图片', async ({ page }) => {
    // 上传 5 张图片
    for (let i = 0; i < 5; i++) {
      const fileInput = page.locator('input[type="file"]')
      await fileInput.setInputFiles(`test-image-${i}.jpg`)
    }
    
    // 验证上传按钮消失
    const uploadButton = page.locator('.ant-upload-select')
    await expect(uploadButton).not.toBeVisible()
  })

  test('删除图片', async ({ page }) => {
    // 上传图片
    const fileInput = page.locator('input[type="file"]')
    await fileInput.setInputFiles('test-image.jpg')
    
    // 删除图片
    const deleteButton = page.locator('.ant-upload-list-item-card .anticon-close')
    await deleteButton.click()
    
    // 验证图片被删除
    const thumbnail = page.locator('.ant-upload-list-picture-card .ant-upload-list-item')
    await expect(thumbnail).not.toBeVisible()
  })
})
```
